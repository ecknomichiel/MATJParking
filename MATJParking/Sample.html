Code samples

Here is a code sample that shows the factory method for creating vehicles: 

private Vehicle CreateVehicle(string aVehicleType)
        {
            switch (aVehicleType)
            {
                case "Motorcycle":
                    return new MotorCycle();
                case "Car":
                    return new Car();
                case "Bus":
                    return new Bus();
                case "Truck":
                    return new Truck();
                default:
                    throw new EUnknownVehicleType(aVehicleType);
            }
        }
